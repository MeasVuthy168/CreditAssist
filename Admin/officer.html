async function uploadToServer() { const fileInput = document.getElementById("uploadImage"); const file = fileInput.files[0]; if (!file) return; const token = sessionStorage.getItem("token"); if (!token) { alert("Session expired. Please log in again."); return; } const formData = new FormData(); formData.append("image", file); try { const res = await fetch("https://secure-backend-tzj9.onrender.com/api/upload", { method: "POST", headers: { Authorization: `Bearer ${token}` // Add token here }, body: formData }); const result = await res.json(); if (res.ok && result.imageUrl) { document.getElementById("previewImage").src = result.imageUrl; document.getElementById("image").value = result.imageUrl; } else { alert("Upload failed: " + (result.message || "Unknown error")); } } catch (err) { console.error("Upload error:", err); alert("Failed to upload image."); } }
